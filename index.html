ğŸ¯ Object-Oriented Programming

class Vehicle {
    let make = "";
    let model = "";
    let year = 0;
    
    fn Vehicle(makeIn, modelIn, yearIn) {
        make = makeIn;
        model = modelIn;
        year = yearIn;
    }
    
    fn toString() {
        return year + " " + make + " " + model;
    }
    
    fn age() {
        return 2025 - year;
    }
}

let car = Vehicle("Toyota", "Camry", 2020);
println(car.toString());  # "2020 Toyota Camry"
println("Age: " + car.age() + " years");

# Fibonacci sequence in Lithium
fn fibonacci(n) {
    if (n <= 1) {
        return n;
    }
    return fibonacci(n - 1) + fibonacci(n - 2);
}

class MathUtils {
    fn factorial(n) {
        if (n <= 1) {
            return 1;
        }
        return n * factorial(n - 1);
    }
}

let utils = MathUtils();
println("Fibonacci(10): " + fibonacci(10));
println("Factorial(5): " + utils.factorial(5));
